快速幂

普通幂运算实现，例如计算 xnx^nxn 的函数。

需要注意的是，当n为负数的时候，需要进行转换，例如2−3=(12)32^{-3} = (\frac{1}{2})^{3}2−3=(21​)3

//x为底数，n为幂
double my_pow(double x,int n){
    double ans{1};
    
    if(n<0){
        x = 1/x;
        n = -n;
    }
    for(int i{0};i<n;i++){
        ans*=x;
    }
    return ans;
}

普通幂需要计算n次乘法。

//x为底数，n为幂
double q_pow(double x,int n){

    double ans{1};
    if(n<0){
        x = 1/x;
        n = -n;
    }
    while(n!=0){
        if(n&1)ans*=x;  //如果当前的位是1，则乘数列的对应项。
        x*=x;  //根据（2）式计算下一项
        n>>=1;
    }
    return ans;
}
